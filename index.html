namespace App\Leads\Application;

final class LeadProcessor
{
    public function __construct(
        private readonly LeadRepository   $leads,
        private readonly EventPublisher   $bus,
        private readonly ScoringService   $scoring,
    ) {}

    public function __invoke(IngestLead $cmd): void
    {
        $lead  = Lead::fromCommand($cmd);
        $score = $this->scoring->evaluate($lead);

        $this->leads->save($lead->withScore($score));
        $this->bus->publish(new LeadIngested($lead->id()));
    }
}
lead-service.js
import { EventBus } from './core/bus.js';
import { Lead }     from './domain/lead.js';

export class LeadService {
  constructor(leads, scoring, bus) {
    this.leads   = leads;
    this.scoring = scoring;
    this.bus     = bus;
  }

  async ingest(command) {
    const lead  = Lead.fromCommand(command);
    const score = await this.scoring.evaluate(lead);

    await this.leads.save(lead.withScore(score));
    await this.bus.publish({ type: 'lead.ingested', id: lead.id });
  }
}
useLeadStatus.ts
import { useEffect, useState } from 'react';

type Status = 'pending' | 'scored' | 'rejected';

export function useLeadStatus(id: string) {
  const [status, setStatus] = useState<Status>('pending');

  useEffect(() => {
    const es = new EventSource(`/api/leads/${id}/events`);

    es.addEventListener('scored',   () => setStatus('scored'));
    es.addEventListener('rejected', () => setStatus('rejected'));

    return () => es.close();
  }, [id]);

  return status;
}
LeadProcessor.java
package app.leads.application;

@Service
public final class LeadProcessor {

  private final LeadRepository leads;
  private final EventPublisher bus;
  private final ScoringService scoring;

  public LeadProcessor(LeadRepository leads,
                        EventPublisher bus,
                        ScoringService scoring) {
    this.leads   = leads;
    this.bus     = bus;
    this.scoring = scoring;
  }

  public void handle(IngestLead cmd) {
    var lead  = Lead.fromCommand(cmd);
    var score = scoring.evaluate(lead);

    leads.save(lead.withScore(score));
    bus.publish(new LeadIngested(lead.id()));
  }
}
