LeadProcessor.php
namespace App\Leads\Application;

final class LeadProcessor
{
    public function __construct(
        private readonly LeadRepository   $leads,
        private readonly EventPublisher   $bus,
        private readonly ScoringService   $scoring,
    ) {}

    public function __invoke(IngestLead $cmd): void
    {
        $lead  = Lead::fromCommand($cmd);
        $score = $this->scoring->evaluate($lead);

        $this->leads->save($lead->withScore($score));
        $this->bus->publish(new LeadIngested($lead->id()));
    }
}
leads.controller.ts
import { Body, Controller, Post, UseGuards } from '@nestjs/common';
import { ApiTags } from '@nestjs/swagger';
import { JwtGuard } from './auth/jwt.guard';
import { LeadsService } from './leads.service';
import { IngestLeadDto } from './dto/ingest-lead.dto';

@ApiTags('leads')
@UseGuards(JwtGuard)
@Controller('leads')
export class LeadsController {
  constructor(private readonly leads: LeadsService) {}

  @Post()
  async ingest(@Body() dto: IngestLeadDto) {
    const lead = await this.leads.ingest(dto);
    return { id: lead.id, score: lead.score };
  }
}
LeadsPage.tsx
import { useQuery } from '@tanstack/react-query';
import { LeadsTable, EmptyState, Spinner } from '@/components';
import { fetchLeads } from '@/api/leads';

export default function LeadsPage() {
  const { data, isLoading, error } = useQuery({
    queryKey: ['leads'],
    queryFn:  fetchLeads,
    staleTime: 30_000,
  });

  if (isLoading) return <Spinner />;
  if (error)     return <EmptyState title="Failed to load" />;

  return (
    <section className="p-6">
      <h1 className="text-2xl font-bold">Leads</h1>
      <p className="mt-1 text-slate-400">{data.length} total</p>
      <LeadsTable rows={data} />
    </section>
  );
}

package app.leads;

import jakarta.ws.rs.*;
import jakarta.ws.rs.core.MediaType;
import io.smallrye.mutiny.Uni;

@Path("/leads")
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
public class LeadResource {

  @Inject LeadService leads;

  @POST
  public Uni<LeadView> ingest(IngestLead cmd) {
    return leads.ingest(cmd)
        .map(LeadView::from);
  }
}
leads_revenue_by_cohort.sql
WITH cohorts AS (
  SELECT id,
         date_trunc('month', created_at) AS cohort
  FROM   leads
  WHERE  status = 'converted'
)
SELECT c.cohort,
       COUNT(DISTINCT c.id)                      AS customers,
       SUM(s.amount)                                 AS mrr,
       ROUND(AVG(s.amount)::numeric, 2)          AS arpu,
       RANK() OVER (ORDER BY SUM(s.amount) DESC) AS rnk
FROM      cohorts       c
JOIN      subscriptions s ON s.lead_id = c.id
WHERE     s.active
GROUP BY  c.cohort
HAVING    COUNT(DISTINCT c.id) > 50
ORDER BY  c.cohort DESC
LIMIT     24;
